#Dictionary functions
marks = {'bhumi': [1,2,3,4,5,], 'Yashwanth' : [6,7,8,9,10]}
#Get all keys
name = marks.keys()
print(name)
#Get all values
name1 = marks.values()
print(name1)
#Get all items
name2 = marks.items()
print(name2)
#All vales must be unique
marks1 = {1,1,2,2,22,3,4,4,4,5,5,56,0,7}
print(marks1)
marks3 = {12, 11, 13, 12, 13, 14, 15,16}
print(marks3)
print(marks1.union(marks3))
